第2步：编写Action类 
package com.edk4j.edk4jdemos.action;

import com.edk4j.core.businessprocessmanage.BusinessAction;
import com.edk4j.core.datastructure.Context;
import com.edk4j.core.exception.TranFailException;
/*
 * 自行在edk4j样例项目中阅读
 * 文件位置：{edk4j样例项目根目录}/src/main/java/com/edk4j/edk4jdemos/action/HelloEdk.java
 */
public class HelloEdk extends BusinessAction{
	@Override
	public String execute(Context context) throws TranFailException {
		try {
			//取得前端输入的字段
			String userName = (String) context.getDataValue("userName");
			//生成欢迎语
			String welcomeWorld = userName+"，欢迎您使用edk4j平台进行开发";
			//将欢迎语设置到context中
			context.setDataValue("welcomeWorld", welcomeWorld);
		}
		catch(TranFailException e) {
			throw e;
		}
		catch(Exception e) {
			throw new TranFailException("xxxxx交易异常", e);
		}
		return OK;
	}
}
