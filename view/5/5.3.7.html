5.3.7、sessionContext.xml
文件位置：${项目路径}/src/main/resources/edk4jConfig/sessionContext.xml
该文件配置sessionContext的原型，即Session数据域定义
样例如下： 
<?xml version="1.0" encoding="UTF-8" ?>
<sessionContext>
	<dataCell id="sessionCustNo" label="登录账号"/>
	<dataCell id="sessionLastLoginTime" label="上次登录时间"/>
	<dataCell id="sessionEmployeeNo" label="工号"/>
	<dataCell id="sessionUserName" label="用户姓名"/>
	<dataCell id="sessionOrgNo" label="所属单位编号"/>
	<dataCell id="sessionDeptNo" label="部门编号"/>
	<dataCell id="sessionApproveType" label="当前身份类型：1-正式员工 2-试用期员工 3-离职员工"/>
	<dataCell id="sessionGradeCode" label="职级编号"/>
	<dataCell id="sessionPostCode" label="岗位编码"/>
	<dataCell id="sessionMobile" label="手机号码"/>
	<dataCell id="sessionCustMenu" label="用户菜单列表，本系统中是个双竖线分割的字符串"/>
	<dataCell id="sessionCustBusiness" label="用户有权操作的交易列表，本系统中是个双竖线分割的字符串"/>
	<dataCell id="sessionNeedChangePassword" label="是否需要登录时修改密码"/>
	...其他dataCell，具体项目中需求并不同...
	<listx id="sessionRoleList" label="用户的角色列表">
		<mapx>
			<dataCell id="roleNo"/>
		</mapx>
	</listx>
       ...其他Listx，具体项目中需求并不同...
</sessionContext>
