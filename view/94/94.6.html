94.6、内置tomcat配置
1、通过配置文件设置
在spring-boot的配置文件中设置：
#系列配置申明于：org.springframework.boot.autoconfigure.web.ServerProperties#Tomcat
#设置tomcat临时文件夹创建的位置
server.tomcat.basedir=/tomcat_temp/
2、编程式配置
生成WebServerFactoryCustomizer实例bean：
.
import java.io.File;
import org.springframework.boot.web.embedded.tomcat.TomcatServletWebServerFactory;
import org.springframework.boot.web.server.WebServerFactoryCustomizer;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
/**
 * 内置tomcat配置
 * @author liuxiaosong(shuidrinking@126.com)
 * @since jdk1.8
 * @version 2024年12月2日
 */
@Configuration
public class TomcatConfig {
	@Bean
	public WebServerFactoryCustomizer tomcatCustomizer() {
		return (tomcatServletWebServerFactory) -> {
			//设置tomcat工作文件夹根目录，每次启动都会在此创建一个新的临时工作文件夹，且不会自动删除
			tomcatServletWebServerFactory.setBaseDirectory(new File("/tomcat_tmp/"));
		};
	}
}
